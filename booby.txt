	<script>
		$('#searchContainerId').keypress( function(event) {
    		if (event.which == '13') 
      			event.preventDefault();
		});

		window.onclick = function (event) {
			if (event.target.matches('#searchButtonId ,#searchId'))
			{	
				$('#searchId').show(300);
				$('#searchContainerId').css({'border-bottom': '3px solid black', 'border-top': '3px solid black'});
				$('#searchButtonId').hide(300);
			}
			else
			{
				$('#searchId').hide(300);
				$('#searchContainerId').css('border-color', 'transparent');
				$('#searchButtonId').show(300);
			}
		};
	</script>
